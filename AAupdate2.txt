-- Add schema_url to levels table to store database diagram image URLs
ALTER TABLE levels ADD COLUMN IF NOT EXISTS schema_url TEXT;
COMMENT ON COLUMN levels.schema_url IS 'URL de la imagen del esquema de base de datos para este nivel';

-- Add diagram_url to questions table for question-specific diagrams
ALTER TABLE questions ADD COLUMN IF NOT EXISTS diagram_url TEXT;
COMMENT ON COLUMN questions.diagram_url IS 'URL de la imagen del esquema de base de datos para esta pregunta';
